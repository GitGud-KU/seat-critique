def valid_floor
  floor = self.floor.to_s
  if !is_number?(floor) && floor.casecmp("L") != 0 && floor.casecmp("LL") != 0 && floor.casecmp("B") != 0
    self.errors[:base] << "Invalid floor"
  end
end
